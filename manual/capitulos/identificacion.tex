\capitulo{Comandos de Identificación}

\noindent Sobre el autor:

\co{autor}\pa, nombre del autor.

\co{citarcomo}\pa, apellido del autor.

\espaciodoble
\noindent Sobre el trabajo:

\co{titulo}\pa, título de la investigación.

\co{title}\pa, título del trabajo en inglés.

\co{palabrasclave}\pa, palabras claves asociadas al trabajo de investigación.

\co{keywords}\pa, palabras claves en ingles.

\co{grado}\pa, permite personalizar el grado al que se desea optar.

\espaciodoble
\noindent Sobre la universidad:

\co{decanato}\pa, decanato de la UCLA.

\co{postgrado}\pa, postgrado cursado.

\espaciodoble
\noindent Sobre el tutor:

\co{tutor}\pa, nombre del tutor.

\espaciodoble
\noindent Sobre la universidad:

\co{ciudad}\pa, ciudad de la defensa.

\co{diadefensa}\pa, dia de la defensa.

\co{mesdefensa}\pa, mes de la defensa.

\co{monthdefensa}\pa, mes de la defensa en inglés para el abstract.

\co{annodefensa}\pa, año de la defensa.

\co{primerjurado}\pa, nombre del primer jurado.

\co{segundojurado}\pa, nombre del segundo jurado.

\espaciodoble
\noindent Opcionales:

\co{mesencaratula}, incluye el mes en las hojas de portada y presentación.

\co{fechaenresumen}, incluye la fecha en las hojas de resumen y abstract.
